# Norse Influx
country_event = {
	id = flavor_ice.3
	title = "flavor_ice.EVTNAME3"
	desc = "flavor_ice.EVTDESC3"
	picture = ADVISOR_eventPicture
	trigger = {
		OR = {
			tag = ICE
			tag = FRO
		}
		has_country_modifier = ice_trade_colonies
		NOT = {
			has_country_flag = vinlanders_turned_away
		}
		NOT = {
			is_year = 1491
		}
		OR = {
			dip = 3
			adm = 3
			mil = 3
		}
	}
	mean_time_to_happen = {
		months = 20
	}
	option = {
		name = "flavor_ice.EVTOPTA3"
		add_adm_power = 10
		add_prestige = 10
		ai_chance = {
			factor = 100
		}
		define_advisor = {
			type = inquisitor
			name = "Magnuz Gunnarsson"
			culture = vinlandic
			religion = norse_pagan_reformed
			skill = 3
			discount = yes
		}
		remove_country_modifier = ice_old_colonies
		add_country_modifier = {
			name = ice_norse_influx
			duration = -1
		}
		add_innovativeness_small_effect = yes
	}
}
